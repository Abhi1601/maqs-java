5   package com.magenic.jmaqs.base;


6   


7   import com.magenic.jmaqs.utilities.logging.Logger;


8   


9   /**


10   * The TestObject class.


11   */


12  public class TestObject {


13    /**


14     * The Logger object.


15     */


16    protected Logger log;


17  


18    /**


19     * The Fully Qualified Test Name.


20     */


21    protected String testName;


22  


23    /**


24     * Initializes a new instance of the TestObject class.


25     * 


26     * @param fullyQualifiedTestName


27     *          The Fully Qualified Test Name


28     */


29    public TestObject(String fullyQualifiedTestName) {


30      this.testName = fullyQualifiedTestName;


31    }


32  


33    /**


34     * Initializes a new instance of the TestObject class.


35     * 


36     * @param fullyQualifiedTestName


37     *          The Fully Qualified Test Name


38     * @param logger


39     *          The Logger object


40     */


41    public TestObject(String fullyQualifiedTestName, Logger logger) {


42      this.testName = fullyQualifiedTestName;


43      this.log = logger;


44    }


45  


46    /**


47     * Get the Fully Qualified Test Name.


48     * 


49     * @return The fully qualified test name


50     */


51    public String getFullyQualifiedTestName() {


52      return testName;


53    }


54  


55    /**


56     * Get the Logger.


57     * 


58     * @return A Logger object


59     */


60    public Logger getLogger() {


61      return this.log;


62    }


63  


64    /**


65     * Set the Logger for the TestObject.


66     * 


67     * @param logger


68     *          The Logger object


69     */


70    public void setLog(Logger logger) {


71      this.log = logger;


72    }


73  }
