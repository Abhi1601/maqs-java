7   import java.lang.reflect.Method;

9   import org.testng.ITestContext;


10  import org.testng.annotations.BeforeMethod;


11  


12  /**


13   * The type Base extendable test.


14   *


15   * @param <T> the type parameter


16   */


17  public abstract class BaseExtendableTest<T extends BaseTestObject> extends BaseTest {


18  


19    /**


20     * Instantiates a new Base extendable test.


21     */


22    protected BaseExtendableTest() {


23    }


24  


25    public T getTestObject() {


26      return (T) super.getTestObject();


27    }


28  


29    @Override


30    public void setTestObject(BaseTestObject baseTestObject) {


31  


32      if (this.baseTestObjects.putIfAbsent(this.getFullyQualifiedTestClassName(), baseTestObject)


33          == null) {


34        this.baseTestObjects.replace(this.getFullyQualifiedTestClassName(), baseTestObject);


35      }


36    }


37  


38    @BeforeMethod


39    @Override


40    public void setup(Method method, ITestContext testContext) throws Exception {


41      super.setup(method, testContext);


42    }


43  


44    protected abstract void createNewTestObject();


45  }
