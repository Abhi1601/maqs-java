189     if (!this.baseTestObjects.containsKey(this.getFullyQualifiedTestClassName())) {


190       this.createNewTestObject();


191     }

193     return this.baseTestObjects.get(this.getFullyQualifiedTestClassName());

196   /**


197    * Sets the Test Object.


198    *


199    * @param baseTestObject The Base Test Object to use


200    */


201   public void setTestObject(BaseTestObject baseTestObject) {


202     String key = this.getFullyQualifiedTestClassName();


203     if (this.baseTestObjects.containsKey(key)) {


204       this.baseTestObjects.replace(key, baseTestObject);


205     } else {


206       this.baseTestObjects.put(key, baseTestObject);


207     }


208   }

210   /**


211    * Setup before a test.


212    *


213    * @param method      The initial executing Method object


214    * @param testContext The initial executing Test Context object


215    * @throws Exception Throws exception if get logger fails


216    */


217   @BeforeMethod(alwaysRun = true)


218   public void setup(Method method, ITestContext testContext) throws Exception {


219     this.testContextInstance = testContext;


220 


221     // Get the Fully Qualified Test Class Name and set it in the object


222     String testName = method.getDeclaringClass() + 
"."
 + method.getName();


223     testName = testName.replaceFirst(
"class "
, 
""
);


224     this.fullyQualifiedTestClassName = testName;


225 


226     this.createNewTestObject();


227   }


228 


229   /**


230    * Cleanup after a test.


231    */


232   @AfterMethod(alwaysRun = true)


233   public void teardown() {


234     try {


235       this.beforeLoggingTeardown(testResult);


236     } catch (Exception e) {


237       this.tryToLog(MessageType.WARNING, 
"Failed before logging teardown because: %s"
,


238           e.getMessage());


239     }


240 


241     // Log the test result


242     if (testResult.getStatus() == ITestResult.SUCCESS) {


243       this.tryToLog(MessageType.SUCCESS, 
"Test Passed"
);


244     } else if (testResult.getStatus() == ITestResult.FAILURE) {


245       this.tryToLog(MessageType.ERROR, 
"Test Failed"
);


246     } else if (testResult.getStatus() == ITestResult.SKIP) {


247       this.tryToLog(MessageType.INFORMATION, 
"Test was skipped"
);


248     } else {


249       this.tryToLog(MessageType.WARNING, 
"Test had an unexpected result."
);


250     }


251 


252     // Cleanup log files we don't want


253     try {


254       if ((this.getLogger() instanceof FileLogger) && testResult.getStatus() == ITestResult.SUCCESS


255           && this.loggingEnabledSetting == LoggingEnabled.ONFAIL) {


256         Files.delete(Paths.get(((FileLogger) this.getLogger()).getFilePath()));


257       }


258     } catch (Exception e) {


259       this.tryToLog(MessageType.WARNING, 
"Failed to cleanup log files because: %s"
, e.getMessage());


260     }


261 


262     // Get the Fully Qualified Test Name


263     String fullyQualifiedTestName = this.getFullyQualifiedTestClassName();


264 


265     // Release logged messages


266     this.loggedExceptions.remove(this.getFullyQualifiedTestClassName());


267 


268     // Release the Base Test Object


269     this.baseTestObjects.remove(fullyQualifiedTestName, this.getTestObject());


270 


271     // Create console logger to log subsequent messages


272     this.setTestObject(new BaseTestObject(new ConsoleLogger(), fullyQualifiedTestName));


273   }

293   /**


294    * Setup logging data.


295    *


296    * @return Logger


297    */


298   protected Logger createLogger() {


299     Logger log;


300 


301     this.loggingEnabledSetting = LoggingConfig.getLoggingEnabledSetting();


302     this.setLoggedExceptions(new ArrayList<String>());


303 


304     if (this.loggingEnabledSetting != LoggingEnabled.NO) {


305       log = LoggingConfig.getLogger(StringProcessor


306           .safeFormatter(
"%s - %s"
, this.getFullyQualifiedTestClassName(),


307               DateTimeFormatter.ofPattern(
"uuuu-MM-dd-HH-mm-ss-SSSS"
, Locale.getDefault())


308                   .format(LocalDateTime.now(Clock.systemUTC()))));


309     } else {


310       log = new ConsoleLogger();


311     }


312 


313     return log;


314   }

321   protected TestResultType getResultType() {


322     switch (this.testResult.getStatus()) {


323       case ITestResult.SUCCESS:


324         return TestResultType.PASS;


325       case ITestResult.FAILURE:


326         return TestResultType.FAIL;


327       case ITestResult.SKIP:


328         return TestResultType.SKIP;


329       default:


330         return TestResultType.OTHER;


331     }


332   }

361   /**


362    * Try to log a message - Do not fail if the message is not logged.


363    *


364    * @param messageType The type of message


365    * @param message     The message text


366    * @param args        String format arguments


367    */


368   protected void tryToLog(MessageType messageType, String message, Object... args) {


369     // Get the formatted message


370     String formattedMessage = StringProcessor.safeFormatter(message, args);


371 


372     try {


373       // Write to the log


374       this.getLogger().logMessage(messageType, formattedMessage);


375 


376       // If this was an error and written to a file, add it to the console


377       // output as well


378       if (messageType == MessageType.ERROR && !(this.getLogger() instanceof ConsoleLogger)) {


379         System.out.println(formattedMessage);


380       }


381     } catch (Exception e) {


382       System.out.println(formattedMessage);


383       System.out.println(
"Logging failed because: "
 + e.getMessage());


384     }


385   }

387   /**


388    * Log a verbose message and include the automation specific call stack data.


389    *


390    * @param message The message text


391    * @param args    String format arguments


392    */


393   protected void logVerbose(String message, Object... args) {


394     StringBuilder messages = new StringBuilder();


395     messages.append(StringProcessor.safeFormatter(message, args) + System.lineSeparator());


396 


397     for (StackTraceElement element : Thread.currentThread().getStackTrace()) {


398       // If the stack trace element is from the com.magenic package (excluding this method) append the stack trace line 


399       if (element.toString().startsWith(
"com.magenic"
) && !element.toString()


400           .contains(
"BaseTest.logVerbose"
)) {


401         messages.append(element.toString() + System.lineSeparator());


402       }


403     }


404 


405     this.getLogger().logMessage(MessageType.VERBOSE, messages.toString());


406   }

412     Logger newLogger = this.createLogger();


413     this.setTestObject(new BaseTestObject(newLogger, this.getFullyQualifiedTestClassName()));
